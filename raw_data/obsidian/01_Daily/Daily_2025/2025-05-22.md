```dataviewjs

// 📆 날짜 설정 - 문서 제목에서 추출
let targetDate;

// 현재 문서의 제목에서 날짜 추출 시도 (yyyy-mm-dd 형식 찾기)
const titleDateMatch = dv.current().file.name.match(/(\d{4}-\d{2}-\d{2})/);

if (titleDateMatch) {
  // 제목에서 날짜를 찾은 경우
  targetDate = titleDateMatch[1];
} else if (dv.current().file.day) {
  // 제목에서 날짜를 찾지 못했지만 day 속성이 있는 경우
  targetDate = dv.current().file.day.toISODate();
} else {
  // 둘 다 없는 경우 오늘 날짜 사용
  const today = new Date();
  targetDate = today.toISOString().split('T')[0];
}

// 디버깅용 날짜 표시
dv.paragraph(`🗓️ **분석 날짜:** ${targetDate}`);

// 시간 포맷 함수 (기존 코드와 동일)
function formatDuration(totalMinutes) {
  if (totalMinutes === 0) return "0m";
  const minutes = Math.round(totalMinutes % 60);
  const hours = Math.floor(totalMinutes / 60);
  let result = "";
  if (hours > 0) result += `${hours}h`;
  if (minutes > 0) {
    if (hours > 0) result += " ";
    result += `${minutes}m`;
  }
  return result;
}

// 해당 날짜의 활동 데이터를 저장할 객체
const dailyActivities = {}; // { activityName: duration }
const activityEnergies = {}; // { activityName: { totalEnergy: value, totalDuration: value } }

// 해당 날짜의 파일 가져오기
const filePath = `01_Daily/Daily/${targetDate}.md`;
const page = dv.page(filePath);

if (!page) {
  dv.paragraph(`❌ **${targetDate}** 파일을 찾을 수 없습니다. 파일 경로를 확인하세요: \`${filePath}\``);
} else {
  // 파일 내용 가져오기
  const fileContent = await dv.io.load(page.file.path);

  // --- 섹션 추출 로직 (기존 코드와 동일) ---
  const plannerHeaderText = "Day planner";
  const headerRegex = new RegExp(`^#+\\s+${plannerHeaderText}\\s*$`, 'mi');
  const headerMatch = fileContent.match(headerRegex);
  let sectionContent = "";

  if (headerMatch) {
    const headerStartIndex = headerMatch.index;
    const matchedHeader = headerMatch[0];
    const contentAfterHeaderStart = fileContent.substring(headerStartIndex + matchedHeader.length);
    const nextHeaderMatch = contentAfterHeaderStart.match(/^\s*#{1,}\s+/m);
    const contentEndIndex = nextHeaderMatch ? nextHeaderMatch.index : contentAfterHeaderStart.length;
    sectionContent = contentAfterHeaderStart.substring(0, contentEndIndex).trim();
  }
  // --- 섹션 추출 로직 끝 ---

  if (sectionContent) {
    const lines = sectionContent.split('\n');
    const lineRegex = /^\s*[-*]\s+(?:\*\*)?(\d{1,2}:\d{2})(?:\*\*)?\s*[-–]\s*(?:\*\*)?(\d{1,2}:\d{2})(?:\*\*)?\s*(.*?)(?:\s+#.*)?$/;

    for (const line of lines) {
      if (!line.trim() || line.trim().startsWith('#')) continue;

      const match = line.trim().match(lineRegex);
      if (match) {
        let startTimeStr = match[1];
        let endTimeStr = match[2];
        let activityName = match[3].trim();

        if (startTimeStr.length === 4) startTimeStr = "0" + startTimeStr;
        if (endTimeStr.length === 4) endTimeStr = "0" + endTimeStr;

        // --- 활동 이름 정리 개선 (기존 코드와 유사하게 적용) ---
        const linkMatches = activityName.match(/!?\[\[(.*?)\]\]/g) || [];
        const tagMatches = line.trim().match(/#[\p{L}\p{N}_]+/gu) || []; // 라인 전체에서 태그 검색
        
        // console.log(`Line: "${line.trim()}", 찾은 태그:`, tagMatches); // 디버깅용

        let mainText = activityName;
        for (const link of linkMatches) {
          mainText = mainText.replace(link, '');
        }
        
        let tags = "";
        for (const tag of tagMatches) {
          mainText = mainText.replace(tag, ''); // 본문에서 태그 제거
          tags += tag + " ";
        }

        mainText = mainText.replace(/^\s*[\p{Emoji_Presentation}\p{Emoji}]\s*/u, '');
        mainText = mainText.replace(/^[ *_~]+|[ *_~]+$/g, '');
        mainText = mainText.replace(/[*_~]{1,3}(.*?)[*_~]{1,3}/g, '$1').trim();

        if (linkMatches.length > 0) {
          mainText = mainText + " " + linkMatches.join(" ");
        }

        activityName = (mainText || "(설명 없음)").trim();

        const energyTag = tagMatches.find(tag => tag.startsWith('#에너지'));
        let energyLevel = 0;
        if (energyTag) {
          const energyMatch = energyTag.match(/#에너지(\d+)/u);
          if (energyMatch && energyMatch[1]) {
            energyLevel = parseInt(energyMatch[1]);
            // console.log(`에너지 태그 발견: ${energyTag}, 추출된 레벨: ${energyLevel}`); // 디버깅용
          } else {
            // console.log(`에너지 값 추출 실패: ${energyTag}`); // 디버깅용
          }
        }
        // --- 활동 이름 정리 끝 ---

        try {
          const start = new Date(`${targetDate}T${startTimeStr}:00`);
          const end = new Date(`${targetDate}T${endTimeStr}:00`);

          if (isNaN(start.getTime()) || isNaN(end.getTime())) {
            console.warn(`유효하지 않은 시간 포맷 건너뜀: ${line.trim()}`);
            continue;
          }

          let adjustedEnd = end;
          if (start >= end) {
             if (startTimeStr.startsWith("2") && endTimeStr.startsWith("0")) {
               adjustedEnd = new Date(`${targetDate}T${endTimeStr}:00`);
               adjustedEnd.setDate(adjustedEnd.getDate() + 1);
             } else {
               console.warn(`종료 시간이 시작 시간보다 빠르거나 같은 항목 건너뜀: ${line.trim()}`);
               continue;
             }
          }

          const duration = (adjustedEnd - start) / (1000 * 60); // 분 단위

          if (duration > 0) {
            // 해당 날짜의 활동 정보 저장
            dailyActivities[activityName] = (dailyActivities[activityName] || 0) + duration;

            // 에너지 데이터 저장
            if (energyLevel > 0) {
              if (!activityEnergies[activityName]) {
                activityEnergies[activityName] = { totalEnergy: 0, totalDuration: 0 };
              }
              // 에너지 * 시간 가중치 누적
              activityEnergies[activityName].totalEnergy += energyLevel * duration;
              activityEnergies[activityName].totalDuration += duration;
            }
          }
        } catch (e) {
          console.error(`시간 처리 오류: ${line.trim()}`, e);
          dv.span(`❌ 시간 처리 오류 발생: ${line.trim()}. 개발자 콘솔(Ctrl+Shift+I) 확인`);
        }
      } else if (line.trim()) {
        // 매치되지 않은 라인 로깅 (디버깅용)
        // console.log(`매치되지 않은 라인: "${line.trim()}"`);
      }
    }
  } else {
    dv.paragraph(`ℹ️ ${targetDate} 파일에서 '${plannerHeaderText}' 섹션을 찾을 수 없거나 내용이 없습니다.`);
  }

  // --- 테이블 생성 로직 ---
  const activityNames = Object.keys(dailyActivities);

  if (activityNames.length === 0) {
    dv.paragraph("📊 해당 날짜에 분석할 활동 데이터가 없습니다.");
  } else {
    // 테이블 헤더 생성
    const headers = ["활동", "시간", "비율", "평균 에너지"];

    // 테이블 데이터 생성
    const tableData = [];
    let totalDayMinutes = 0;

    // 해당 날짜의 총 시간 계산
    for (const activity of activityNames) {
      totalDayMinutes += dailyActivities[activity];
    }

    // 각 활동별 데이터 생성
    for (const activity of activityNames) {
      const duration = dailyActivities[activity];
      const formattedDuration = formatDuration(duration);

      // 퍼센트 계산
      const percentage = totalDayMinutes > 0 ? Math.round((duration / totalDayMinutes) * 100) : 0;

      // 평균 에너지 계산
      let avgEnergy = "-";
      if (activityEnergies[activity] && activityEnergies[activity].totalDuration > 0) {
        // 해당 활동의 총 에너지 가중 합계를 총 시간으로 나누어 평균 계산
        const energyAvg = Math.round(activityEnergies[activity].totalEnergy / activityEnergies[activity].totalDuration);
        avgEnergy = `${energyAvg}`;
      }

      tableData.push([
        activity,
        formattedDuration,
        `${percentage}%`,
        avgEnergy
      ]);
    }

    // 테이블 데이터 정렬 (시간 기준 내림차순)
    tableData.sort((a, b) => {
      const getMinutes = (timeStr) => {
        if (timeStr === "-" || timeStr === "0m") return 0;
        let totalMinutes = 0;
        const hourMatch = timeStr.match(/(\d+)h/);
        const minuteMatch = timeStr.match(/(\d+)m/);
        if (hourMatch) totalMinutes += parseInt(hourMatch[1]) * 60;
        if (minuteMatch) totalMinutes += parseInt(minuteMatch[1]);
        return totalMinutes;
      };
      // 시간은 두 번째 열(index 1)
      return getMinutes(b[1]) - getMinutes(a[1]);
    });

    // 테이블 Markdown 생성
    let table = `| ${headers.join(" | ")} |\n`;
    table += `| ${headers.map(_ => "---").join(" | ")} |\n`;
    for (const row of tableData) {
      table += `| ${row.join(" | ")} |\n`;
    }

    // 테이블 출력
    dv.paragraph(table);

    // --- 요약 정보 계산 및 출력 ---
    let totalEnergyWeighted = 0;
    let totalEnergyDuration = 0;

    // 해당 날짜의 전체 평균 에너지 계산
    for (const activity in activityEnergies) {
        totalEnergyWeighted += activityEnergies[activity].totalEnergy;
        totalEnergyDuration += activityEnergies[activity].totalDuration;
    }

    const overallAvgEnergy = totalEnergyDuration > 0 ?
      Math.round(totalEnergyWeighted / totalEnergyDuration) : 0;

    dv.paragraph(`**총 기록 시간:** ${formatDuration(totalDayMinutes)} | **평균 에너지:** ${overallAvgEnergy}`);
  }
}
```
