  const activityNames = Object.keys(dailyActivities);

  if (activityNames.length === 0) {
    dv.paragraph("📊 해당 날짜에 분석할 활동 데이터가 없습니다.");
  } else {
    // 테이블 헤더 생성
    const headers = ["활동", "시간", "비율", "평균 에너지"];

    // 테이블 데이터 생성
    const tableData = [];
    let totalDayMinutes = 0;

    // 해당 날짜의 총 시간 계산
    for (const activity of activityNames) {
      totalDayMinutes += dailyActivities[activity];
    }

    // 각 활동별 데이터 생성
    for (const activity of activityNames) {
      const duration = dailyActivities[activity];
      const formattedDuration = formatDuration(duration);

      // 퍼센트 계산
      const percentage = totalDayMinutes > 0 ? Math.round((duration / totalDayMinutes) * 100) : 0;

      // 평균 에너지 계산
      let avgEnergy = "-";
      if (activityEnergies[activity] && activityEnergies[activity].totalDuration > 0) {
        // 해당 활동의 총 에너지 가중 합계를 총 시간으로 나누어 평균 계산
        const energyAvg = Math.round(activityEnergies[activity].totalEnergy / activityEnergies[activity].totalDuration);
        avgEnergy = `${energyAvg}`;
      }

      tableData.push([
        activity,
        formattedDuration,
        `${percentage}%`,
        avgEnergy
      ]);
    }

    // 테이블 데이터 정렬 (시간 기준 내림차순)
    tableData.sort((a, b) => {
      const getMinutes = (timeStr) => {
        if (timeStr === "-" || timeStr === "0m") return 0;
        let totalMinutes = 0;
        const hourMatch = timeStr.match(/(\d+)h/);
        const minuteMatch = timeStr.match(/(\d+)m/);
        if (hourMatch) totalMinutes += parseInt(hourMatch[1]) * 60;
        if (minuteMatch) totalMinutes += parseInt(minuteMatch[1]);
        return totalMinutes;
      };
      // 시간은 두 번째 열(index 1)
      return getMinutes(b[1]) - getMinutes(a[1]);
    });

    // 테이블 Markdown 생성
    let table = `| ${headers.join(" | ")} |\n`;
    table += `| ${headers.map(_ => "---").join(" | ")} |\n`;
    for (const row of tableData) {
      table += `| ${row.join(" | ")} |\n`;
    }

    // 테이블 출력
    dv.paragraph(table);

    // --- 요약 정보 계산 및 출력 ---
    let totalEnergyWeighted = 0;
    let totalEnergyDuration = 0;

    // 해당 날짜의 전체 평균 에너지 계산
    for (const activity in activityEnergies) {
        totalEnergyWeighted += activityEnergies[activity].totalEnergy;
        totalEnergyDuration += activityEnergies[activity].totalDuration;
    }

    const overallAvgEnergy = totalEnergyDuration > 0 ?
      Math.round(totalEnergyWeighted / totalEnergyDuration) : 0;

    dv.paragraph(`**총 기록 시간:** ${formatDuration(totalDayMinutes)} | **평균 에너지:** ${overallAvgEnergy}`);
  }
}
