import math

```python

# --- PCIe Constants and Parameters ---
PCIE_GEN = 5
NUM_LANES = 4
TOTAL_DATA_KB = 128

# Addressing and other assumptions (can be modified if needed)
ADDRESSING_BITS = 32 # Assume 32-bit for header size calculation
USE_ECRC = False
USE_TLP_PREFIX = False
CLOCKING_MODE = "SRNS" # "SRNS" or "SRIS". Affects SKP interval

# Encoding details (Gen3+)
ENCODING_PAYLOAD_BITS = 128
ENCODING_TOTAL_BITS = 130
BLOCK_BITS = ENCODING_TOTAL_BITS # 128b/130b block size

# Conversion factors
BYTES_PER_KB = 1024
BITS_PER_BYTE = 8
BITS_PER_GBPS = 1_000_000_000 # 1 Giga-bit per second

# --- Lane Speed based on Gen ---
if PCIE_GEN == 5:
    LANE_SPEED_GTPS = 32
elif PCIE_GEN == 4:
    LANE_SPEED_GTPS = 16
elif PCIE_GEN == 3:
    LANE_SPEED_GTPS = 8
else:
    raise ValueError("Unsupported PCIe Generation specified")

LANE_SPEED_GBPS = LANE_SPEED_GTPS # GT/s directly translates to Gbps for PCIe

# --- Header Sizes (Bytes) ---
# Assuming MWr Request
if ADDRESSING_BITS == 32:
    TL_HEADER_BYTES_NFM = 3 * 4 # 3 DW header
    # FM Header Base Size can vary, but for MWr 32b it's 3DW base
    TL_HEADER_BASE_BYTES_FM = 3 * 4
elif ADDRESSING_BITS == 64:
    TL_HEADER_BYTES_NFM = 4 * 4 # 4 DW header
    # FM Header Base Size for MWr 64b it's 4DW base
    TL_HEADER_BASE_BYTES_FM = 4 * 4
else:
     raise ValueError("Unsupported Addressing Bits specified")

# --- DLL Overhead (NFM only) ---
DLL_OVERHEAD_BYTES_NFM = 2 + 4 # 16b Seq Num + 32b LCRC

# --- Flit Details ---
FLIT_TOTAL_BYTES = 256
FLIT_TLP_PAYLOAD_BYTES = 236
FLIT_DLP_BYTES = 6
FLIT_CRC_BYTES = 8
FLIT_ECC_BYTES = 6
FLIT_OVERHEAD_BYTES = FLIT_DLP_BYTES + FLIT_CRC_BYTES + FLIT_ECC_BYTES # = 20 Bytes

# --- SKP OS Interval Parameters ---
# Non-Flit Mode (NFM) - 128b/130b
SKP_INTERVAL_BLOCKS_NFM_SRNS = 372.5 # Average for SRNS/Common Clock
SKP_INTERVAL_BLOCKS_NFM_SRIS = 37.5 # Midpoint of < 38 Blocks
SKP_OS_BITS_NFM = BLOCK_BITS # SKP OS is one block
EDS_OVERHEAD_BITS_NFM = 32 # Replaces data in the last DW before SKP OS

# Flit Mode (FM) - 128b/130b, x4 Lane (from Table 4-30)
SKP_INTERVAL_FLITS_FM_SRNS = 93
SKP_INTERVAL_FLITS_FM_SRIS = 9
SKP_OS_BITS_FM = BLOCK_BITS # SKP OS is still one block

# --- Function to Calculate Throughput ---
def calculate_pcie_throughput(mps_bytes):
    """Calculates theoretical PCIe throughput for given MPS."""
    print(f"\n--- Calculating for MPS = {mps_bytes} Bytes ---")

    total_data_bytes = TOTAL_DATA_KB * BYTES_PER_KB

    # --- Phase 1: Raw Link Bandwidth ---
    total_raw_bw_gbps = LANE_SPEED_GBPS * NUM_LANES

    # --- Phase 2: Encoding Overhead ---
    encoding_efficiency = ENCODING_PAYLOAD_BITS / ENCODING_TOTAL_BITS
    effective_link_bw_gbps = total_raw_bw_gbps * encoding_efficiency
    effective_link_bw_GBps = effective_link_bw_gbps / BITS_PER_BYTE
    print(f"1. Effective Link Bandwidth (after 128b/130b encoding): {effective_link_bw_GBps:.3f} GB/s ({effective_link_bw_gbps:.3f} Gbps)")

    # --- Non-Flit Mode (NFM) Calculation ---
    print("\n-- Non-Flit Mode (NFM) --")
    # Phase 3: TLP Overhead (NFM)
    num_tlps_nfm = math.ceil(total_data_bytes / mps_bytes)
    tlp_overhead_bytes_per_tlp_nfm = TL_HEADER_BYTES_NFM + DLL_OVERHEAD_BYTES_NFM
    total_tlp_overhead_bytes_nfm = tlp_overhead_bytes_per_tlp_nfm * num_tlps_nfm
    total_tlp_tx_bytes_nfm = total_data_bytes + total_tlp_overhead_bytes_nfm
    print(f"   - TLPs needed: {num_tlps_nfm}")
    print(f"   - Overhead per TLP (Header + DLL): {tlp_overhead_bytes_per_tlp_nfm} Bytes")
    print(f"   - Total TLP Overhead: {total_tlp_overhead_bytes_nfm} Bytes")
    print(f"   - Total Bytes for TLPs (Payload + Overhead): {total_tlp_tx_bytes_nfm} Bytes")

    # Phase 4: Physical Layer Overhead (NFM - SKP OS + EDS)
    if CLOCKING_MODE == "SRIS":
        skp_interval_blocks_nfm = SKP_INTERVAL_BLOCKS_NFM_SRIS
    else: # SRNS or Common Clock
        skp_interval_blocks_nfm = SKP_INTERVAL_BLOCKS_NFM_SRNS

    total_bits_per_cycle_nfm = skp_interval_blocks_nfm * BLOCK_BITS
    # EDS replaces 32 bits within a data block just before SKP OS block
    overhead_bits_per_cycle_nfm = SKP_OS_BITS_NFM + EDS_OVERHEAD_BITS_NFM
    physical_overhead_ratio_nfm = overhead_bits_per_cycle_nfm / total_bits_per_cycle_nfm
    print(f"   - Physical Layer Overhead Ratio (SKP OS + EDS, {CLOCKING_MODE}): {physical_overhead_ratio_nfm:.4f} ({physical_overhead_ratio_nfm*100:.2f}%)")

    # Phase 5: Effective Throughput (NFM)
    available_bw_after_phys_overhead_nfm = effective_link_bw_GBps * (1 - physical_overhead_ratio_nfm)
    tlp_efficiency_nfm = total_data_bytes / total_tlp_tx_bytes_nfm
    effective_throughput_nfm_GBps = available_bw_after_phys_overhead_nfm * tlp_efficiency_nfm
    print(f"   - TLP Efficiency (Payload / TLP Total Bytes): {tlp_efficiency_nfm:.4f} ({tlp_efficiency_nfm*100:.2f}%)")
    print(f"   - >>> Effective NFM Throughput: {effective_throughput_nfm_GBps:.3f} GB/s <<<")
    print(f"   - Theoretical NFM transfer time for {TOTAL_DATA_KB}KB: {(total_data_bytes / (effective_throughput_nfm_GBps * 1024**3)) * 1_000_000:.2f} µs")

    # --- Flit Mode (FM) Calculation ---
    print("\n-- Flit Mode (FM) --")
    # Flit overhead calculation
    flit_payload_bits = FLIT_TLP_PAYLOAD_BYTES * BITS_PER_BYTE
    flit_total_bits = FLIT_TOTAL_BYTES * BITS_PER_BYTE
    flit_tx_bits_encoded = flit_total_bits * ENCODING_TOTAL_BITS / ENCODING_PAYLOAD_BITS
    flit_total_tx_bytes_encoded = flit_tx_bits_encoded / BITS_PER_BYTE
    flit_efficiency_fm = FLIT_TLP_PAYLOAD_BYTES / flit_total_tx_bytes_encoded
    print(f"   - Flit Efficiency (TLP Payload / Encoded Flit Bytes): {flit_efficiency_fm:.4f} ({flit_efficiency_fm*100:.2f}%)")

    # Phase 6: Physical Layer Overhead (FM - SKP OS only)
    if CLOCKING_MODE == "SRIS":
        skp_interval_flits_fm = SKP_INTERVAL_FLITS_FM_SRIS
    else: # SRNS or Common Clock
        skp_interval_flits_fm = SKP_INTERVAL_FLITS_FM_SRNS

    total_bits_per_cycle_fm = skp_interval_flits_fm * flit_tx_bits_encoded
    overhead_bits_per_cycle_fm = SKP_OS_BITS_FM # SKP OS uses one block (130 bits)
    physical_overhead_ratio_fm = overhead_bits_per_cycle_fm / total_bits_per_cycle_fm
    print(f"   - Physical Layer Overhead Ratio (SKP OS, {CLOCKING_MODE}): {physical_overhead_ratio_fm:.4f} ({physical_overhead_ratio_fm*100:.2f}%)")

    # Phase 7 & 8: Effective Throughput (FM)
    available_bw_after_phys_overhead_fm = effective_link_bw_GBps * (1 - physical_overhead_ratio_fm)
    effective_throughput_fm_GBps = available_bw_after_phys_overhead_fm * flit_efficiency_fm
    print(f"   - >>> Effective FM Throughput: {effective_throughput_fm_GBps:.3f} GB/s <<<")
    print(f"   - Theoretical FM transfer time for {TOTAL_DATA_KB}KB: {(total_data_bytes / (effective_throughput_fm_GBps * 1024**3)) * 1_000_000:.2f} µs")

# --- Run Calculations for both MPS values ---
mps_values = [128, 512]
for mps in mps_values:
    calculate_pcie_throughput(mps)

print("\n--- Notes ---")
print(f"* Calculation for {TOTAL_DATA_KB}KB Memory Write on PCIe Gen{PCIE_GEN} x{NUM_LANES}.")
print(f"* Assumes {ADDRESSING_BITS}-bit addressing, no ECRC, no TLP Prefixes.")
print(f"* Assumes ideal {CLOCKING_MODE} clocking (lower SKP overhead).")
print("* Assumes NO Flow Control delays, NO Ack/Nak latency, NO errors/retries.")
print("* This is a THEORETICAL maximum throughput calculation.")
print("* Real-world performance will be lower due to factors not included here.")
print("* NFM calculation now explicitly includes EDS token overhead.")
print("* FM calculation efficiency is dominated by the fixed Flit structure.")
```

```python

import math

# --- PCIe Constants and Parameters ---
PCIE_GEN = 5
NUM_LANES = 4
TOTAL_DATA_KB = 128

# Addressing and header assumptions
ADDRESSING_BITS = 32  # 32-bit addressing for header size
USE_ECRC = False
USE_TLP_PREFIX = False
CLOCKING_MODE = "SRNS"  # "SRNS" or "SRIS"

# Encoding details (Gen3+ uses 128b/130b)
ENCODING_PAYLOAD_BITS = 128
ENCODING_TOTAL_BITS = 130
BLOCK_BITS = ENCODING_TOTAL_BITS

# Conversion factors
BYTES_PER_KB = 1024
BITS_PER_BYTE = 8
BITS_PER_GBPS = 1_000_000_000

# --- Lane Speed based on PCIe Generation ---
if PCIE_GEN == 5:
    LANE_SPEED_GTPS = 32
elif PCIE_GEN == 4:
    LANE_SPEED_GTPS = 16
elif PCIE_GEN == 3:
    LANE_SPEED_GTPS = 8
else:
    raise ValueError("Unsupported PCIe Generation")

LANE_SPEED_GBPS = LANE_SPEED_GTPS  # For PCIe, GT/s ≈ Gbps

# --- Header Sizes (Bytes) ---
if ADDRESSING_BITS == 32:
    TL_HEADER_BYTES_NFM = 3 * 4  # 3 DW header
    TL_HEADER_BASE_BYTES_FM = 3 * 4
elif ADDRESSING_BITS == 64:
    TL_HEADER_BYTES_NFM = 4 * 4  # 4 DW header
    TL_HEADER_BASE_BYTES_FM = 4 * 4
else:
    raise ValueError("Unsupported Addressing Bits")

DLL_OVERHEAD_BYTES_NFM = 2 + 4  # 16b Seq Num + 32b LCRC

# --- Flit Details ---
FLIT_TOTAL_BYTES = 256
FLIT_TLP_PAYLOAD_BYTES = 236
FLIT_DLP_BYTES = 6
FLIT_CRC_BYTES = 8
FLIT_ECC_BYTES = 6
FLIT_OVERHEAD_BYTES = FLIT_DLP_BYTES + FLIT_CRC_BYTES + FLIT_ECC_BYTES  # 20 Bytes

# --- SKP OS Interval Parameters ---
SKP_INTERVAL_BLOCKS_NFM = 372.5 if CLOCKING_MODE == "SRNS" else 37.5
SKP_OS_BITS_NFM = BLOCK_BITS
EDS_OVERHEAD_BITS_NFM = 32

SKP_INTERVAL_FLITS_FM = 93 if CLOCKING_MODE == "SRNS" else 9
SKP_OS_BITS_FM = BLOCK_BITS

# --- Throughput Calculation Function ---
def calculate_pcie_throughput(mps_bytes):
    print(f"\n--- Calculating for MPS = {mps_bytes} Bytes ---")

    total_data_bytes = TOTAL_DATA_KB * BYTES_PER_KB
    total_raw_bw_gbps = LANE_SPEED_GBPS * NUM_LANES
    encoding_efficiency = ENCODING_PAYLOAD_BITS / ENCODING_TOTAL_BITS
    effective_link_bw_gbps = total_raw_bw_gbps * encoding_efficiency
    effective_link_bw_GBps = effective_link_bw_gbps / BITS_PER_BYTE

    print(f"1. Effective Link Bandwidth: {effective_link_bw_GBps:.3f} GB/s ({effective_link_bw_gbps:.3f} Gbps)")

    # --- Non-Flit Mode ---
    print("\n-- Non-Flit Mode (NFM) --")
    num_tlps = math.ceil(total_data_bytes / mps_bytes)
    tlp_overhead_bytes = TL_HEADER_BYTES_NFM + DLL_OVERHEAD_BYTES_NFM
    total_tlp_tx_bytes = total_data_bytes + num_tlps * tlp_overhead_bytes

    physical_overhead_ratio = (SKP_OS_BITS_NFM + EDS_OVERHEAD_BITS_NFM) / (SKP_INTERVAL_BLOCKS_NFM * BLOCK_BITS)
    available_bw_nfm = effective_link_bw_GBps * (1 - physical_overhead_ratio)
    tlp_efficiency = total_data_bytes / total_tlp_tx_bytes
    throughput_nfm = available_bw_nfm * tlp_efficiency
    transfer_time_nfm_us = (total_data_bytes / (throughput_nfm * 1024 ** 3)) * 1_000_000

    print(f"   - TLP Count: {num_tlps}")
    print(f"   - TLP Overhead per TLP: {tlp_overhead_bytes} Bytes")
    print(f"   - Total Overhead: {num_tlps * tlp_overhead_bytes} Bytes")
    print(f"   - Physical Overhead Ratio: {physical_overhead_ratio:.4f} ({physical_overhead_ratio * 100:.2f}%)")
    print(f"   - TLP Efficiency: {tlp_efficiency:.4f} ({tlp_efficiency * 100:.2f}%)")
    print(f"   - >>> Effective NFM Throughput: {throughput_nfm:.3f} GB/s")
    print(f"   - Theoretical Transfer Time: {transfer_time_nfm_us:.2f} µs")

    # --- Flit Mode ---
    print("\n-- Flit Mode (FM) --")
    flit_payload_bits = FLIT_TLP_PAYLOAD_BYTES * BITS_PER_BYTE
    flit_total_bits = FLIT_TOTAL_BYTES * BITS_PER_BYTE
    flit_encoded_bits = flit_total_bits * ENCODING_TOTAL_BITS / ENCODING_PAYLOAD_BITS
    flit_encoded_bytes = flit_encoded_bits / BITS_PER_BYTE
    flit_efficiency = FLIT_TLP_PAYLOAD_BYTES / flit_encoded_bytes

    physical_overhead_ratio_fm = SKP_OS_BITS_FM / (SKP_INTERVAL_FLITS_FM * flit_encoded_bits)
    available_bw_fm = effective_link_bw_GBps * (1 - physical_overhead_ratio_fm)
    throughput_fm = available_bw_fm * flit_efficiency
    transfer_time_fm_us = (total_data_bytes / (throughput_fm * 1024 ** 3)) * 1_000_000

    print(f"   - Flit Efficiency: {flit_efficiency:.4f} ({flit_efficiency * 100:.2f}%)")
    print(f"   - Physical Overhead Ratio: {physical_overhead_ratio_fm:.4f} ({physical_overhead_ratio_fm * 100:.2f}%)")
    print(f"   - >>> Effective FM Throughput: {throughput_fm:.3f} GB/s")
    print(f"   - Theoretical Transfer Time: {transfer_time_fm_us:.2f} µs")

# --- Run Throughput for Example MPS Values ---
for mps in [128, 512]:
    calculate_pcie_throughput(mps)

# --- Notes ---
print("\n--- Notes ---")
print(f"* Calculation for {TOTAL_DATA_KB}KB MWr on PCIe Gen{PCIE_GEN} x{NUM_LANES}")
print(f"* {ADDRESSING_BITS}-bit addressing, No ECRC, No TLP Prefix")
print(f"* Clocking Mode: {CLOCKING_MODE}")
print("* Ideal conditions assumed: No errors, flow control, Ack/Nak, etc.")
print("* Real-world performance will be lower.")
```
